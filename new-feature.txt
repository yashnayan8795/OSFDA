"""
Manual Test Data Upload — shared sidebar component.

Renders on every page (app.py + each page in pages/). Lets the user pick a
problem, see the required data format, and upload a CSV/Parquet. On
successful upload, stashes the dataframe in session_state and navigates
to the corresponding problem page.

Session state keys used:
    mt_uploaded_dfs:        dict[problem_id, pd.DataFrame]
                            — uploaded data, keyed by problem
    mt_pending_navigation:  str | None
                            — set to a page path right after a successful
                              upload; consumed once and cleared so we don't
                              loop on rerun
    mt_last_upload_id:      str | None
                            — the file_id of the last consumed upload, so
                              we don't re-trigger navigation on the next
                              rerun while the same file is still in the
                              uploader's state

The page targets are mapped here in one place so renaming a page only
requires editing this file.
"""
from __future__ import annotations

import streamlit as st

from utils.schemas import SCHEMAS, UNSUPPORTED_PROBLEMS, get_schema
from utils.validation import read_upload, validate, MAX_ROWS

# Map problem_id -> the Streamlit page path used by st.switch_page.
# Adjust if the page filenames change.
PAGE_PATHS: dict[str, str] = {
    "severity":  "pages/1_Problem_A_Severity.py",
    "preflight": "pages/3_Problem_C_Preflight.py",
}


def _init_state() -> None:
    if "mt_uploaded_dfs" not in st.session_state:
        st.session_state.mt_uploaded_dfs = {}
    if "mt_pending_navigation" not in st.session_state:
        st.session_state.mt_pending_navigation = None
    if "mt_last_upload_id" not in st.session_state:
        st.session_state.mt_last_upload_id = None


def get_uploaded_df(problem_id: str):
    """Page-side accessor — returns the uploaded dataframe for this problem, or None."""
    _init_state()
    return st.session_state.mt_uploaded_dfs.get(problem_id)


def clear_uploaded_df(problem_id: str) -> None:
    """Allow a page to clear its uploaded data (e.g. via a 'reset' button)."""
    _init_state()
    st.session_state.mt_uploaded_dfs.pop(problem_id, None)


def _consume_pending_navigation() -> None:
    """If a navigation was queued by a previous rerun, fire it now and clear."""
    target = st.session_state.mt_pending_navigation
    if target:
        st.session_state.mt_pending_navigation = None
        try:
            st.switch_page(target)
        except Exception:
            # switch_page can fail if the path is wrong; surface gently.
            st.sidebar.warning(f"Could not navigate to `{target}`. Open the page manually.")


def _render_format_help(problem_id: str) -> None:
    schema = get_schema(problem_id)
    if schema is None:
        st.info(
            f"Manual upload isn't supported yet for "
            f"**{UNSUPPORTED_PROBLEMS.get(problem_id, problem_id)}**. "
            f"Pick Problem A or Problem C."
        )
        return

    st.caption(schema.notes)

    with st.expander("Required columns (strict)", expanded=False):
        for col in schema.required:
            st.markdown(f"- `{col}`")

    if schema.optional:
        with st.expander("Optional columns (filled with NaN if missing)", expanded=False):
            for col in schema.optional:
                st.markdown(f"- `{col}`")

    if schema.target:
        st.caption(
            f"📊 Include a `{schema.target}` column for evaluation metrics. "
            f"Without it, only predictions will be shown."
        )


def render_manual_test_sidebar() -> None:
    """
    Render the Manual Test section in the sidebar. Call this from every page.

    Idempotent — safe to call multiple times per rerun (Streamlit will dedupe
    via the unique widget keys).
    """
    _init_state()

    # Fire any queued navigation BEFORE rendering, so we don't waste a render.
    _consume_pending_navigation()

    with st.sidebar:
        st.divider()
        with st.expander("🧪 Manual Test", expanded=False):
            st.caption(
                "Upload your own dataset to run any of the supported models. "
                f"Max **{MAX_ROWS:,} rows**, CSV or Parquet."
            )

            # --- problem selector ------------------------------------------
            options = []
            for pid, schema in SCHEMAS.items():
                options.append((pid, schema.display_name))
            for pid, name in UNSUPPORTED_PROBLEMS.items():
                options.append((pid, f"{name} (not yet supported)"))

            labels = [name for _, name in options]
            ids = [pid for pid, _ in options]

            choice_idx = st.selectbox(
                "Problem",
                range(len(options)),
                format_func=lambda i: labels[i],
                key="mt_problem_select",
            )
            problem_id = ids[choice_idx]

            # --- format instructions ---------------------------------------
            _render_format_help(problem_id)

            # Don't show the uploader for unsupported problems.
            if problem_id not in SCHEMAS:
                return

            # --- file uploader ---------------------------------------------
            uploaded = st.file_uploader(
                "Upload CSV or Parquet",
                type=["csv", "parquet"],
                key=f"mt_upload_{problem_id}",
                help=(
                    "The file is parsed in-memory and validated against the "
                    "schema above. Required columns must be present."
                ),
            )

            # Show currently active upload, if any
            existing = st.session_state.mt_uploaded_dfs.get(problem_id)
            if existing is not None and uploaded is None:
                st.success(
                    f"✅ Active dataset: {len(existing):,} rows × "
                    f"{len(existing.columns)} columns"
                )
                if st.button("Clear uploaded data", key=f"mt_clear_{problem_id}"):
                    clear_uploaded_df(problem_id)
                    st.session_state.mt_last_upload_id = None
                    st.rerun()
                return

            if uploaded is None:
                return

            # The uploader keeps returning the same file across reruns; only
            # process it once per file_id.
            file_id = getattr(uploaded, "file_id", None) or uploaded.name
            already_consumed = (
                st.session_state.mt_last_upload_id == file_id
                and existing is not None
            )
            if already_consumed:
                return

            # --- read + validate -------------------------------------------
            df, read_err = read_upload(uploaded)
            if read_err:
                st.error(read_err)
                return

            schema = get_schema(problem_id)
            result = validate(df, schema)

            if not result.ok:
                for msg in result.errors:
                    st.error(msg)
                return

            for w in result.warnings:
                st.warning(w)

            # --- commit + queue navigation ---------------------------------
            st.session_state.mt_uploaded_dfs[problem_id] = result.df
            st.session_state.mt_last_upload_id = file_id
            st.success(
                f"✅ Loaded {len(result.df):,} rows. Switching to "
                f"{schema.display_name}…"
            )

            target = PAGE_PATHS.get(problem_id)
            if target:
                st.session_state.mt_pending_navigation = target
                st.rerun()

Single source of truth: the validator AND the on-screen format instructions

Each schema declares:
    required:  columns that MUST be present (strict)

    target:    the ground-truth label column, if any (used for evaluation

"""
Validation for manually uploaded test data.
 
Policy (per agreed design):
  - REQUIRED columns: strict. If any are missing, validation fails.
  - OPTIONAL columns: lenient. Missing ones are added as NaN with a warning.
  - Row cap: 10,000. Above that, fail with a clear message.
 
Returns a structured result so the sidebar can render errors/warnings
and the page can know whether to proceed.
"""
from __future__ import annotations
 
from dataclasses import dataclass, field
 
import pandas as pd
 
from utils.schemas import UploadSchema
 
MAX_ROWS = 10_000
 
 
@dataclass
class ValidationResult:
    ok: bool
    df: pd.DataFrame | None = None
    errors: list[str] = field(default_factory=list)
    warnings: list[str] = field(default_factory=list)
 
 
def read_upload(uploaded_file) -> tuple[pd.DataFrame | None, str | None]:
    """Read a CSV or Parquet upload into a DataFrame. Returns (df, error)."""
    name = (uploaded_file.name or "").lower()
    try:
        if name.endswith(".csv"):
            df = pd.read_csv(uploaded_file)
        elif name.endswith(".parquet") or name.endswith(".pq"):
            df = pd.read_parquet(uploaded_file)
        else:
            return None, f"Unsupported file type: `{uploaded_file.name}`. Use .csv or .parquet."
        return df, None
    except Exception as e:
        return None, f"Could not read `{uploaded_file.name}`: {e}"
 
 
def validate(df: pd.DataFrame, schema: UploadSchema) -> ValidationResult:
    """Run strict-required + lenient-optional validation against a schema."""
    errors: list[str] = []
    warnings: list[str] = []
 
    # --- row cap ------------------------------------------------------------
    if len(df) > MAX_ROWS:
        errors.append(
            f"Upload has {len(df):,} rows; the limit is {MAX_ROWS:,}. "
            f"Trim the file or sample down before uploading."
        )
        return ValidationResult(ok=False, errors=errors, warnings=warnings)
 
    if len(df) == 0:
        errors.append("Upload is empty (0 rows).")
        return ValidationResult(ok=False, errors=errors, warnings=warnings)
 
    # --- strict: required columns ------------------------------------------
    missing_required = [c for c in schema.required if c not in df.columns]
    if missing_required:
        errors.append(
            "Missing required columns: "
            + ", ".join(f"`{c}`" for c in missing_required)
        )
        return ValidationResult(ok=False, errors=errors, warnings=warnings)
 
    # --- lenient: optional columns -----------------------------------------
    df = df.copy()
    missing_optional = [c for c in schema.optional if c not in df.columns]
    if missing_optional:
        for c in missing_optional:
            df[c] = pd.NA
        warnings.append(
            f"{len(missing_optional)} optional column(s) missing — filled with NaN: "
            + ", ".join(f"`{c}`" for c in missing_optional[:6])
            + ("…" if len(missing_optional) > 6 else "")
        )
 
    # --- dtype coercion (best-effort, non-fatal) ---------------------------
    for col, dtype in schema.dtypes.items():
        if col not in df.columns:
            continue
        try:
            if dtype.startswith("datetime"):
                df[col] = pd.to_datetime(df[col], errors="coerce")
            else:
                df[col] = df[col].astype(dtype, errors="ignore")
        except Exception as e:
            warnings.append(f"Could not coerce `{col}` to {dtype}: {e}")
 
    # --- target column note (informational) --------------------------------
    if schema.target and schema.target not in df.columns:
        warnings.append(
            f"No `{schema.target}` column found — predictions will be shown "
            f"without evaluation metrics."
        )
 
    # --- unknown columns are fine, just note them --------------------------
    known = set(schema.all_columns) | ({schema.target} if schema.target else set())
    extras = [c for c in df.columns if c not in known]
    if extras:
        warnings.append(
            f"{len(extras)} extra column(s) ignored by the model: "
            + ", ".join(f"`{c}`" for c in extras[:6])
            + ("…" if len(extras) > 6 else "")
        )
 
    return ValidationResult(ok=True, df=df, errors=errors, warnings=warnings)

def _align_columns(df: pd.DataFrame, expected: list[str]) -> pd.DataFrame:
    """Reorder columns to match training-time order; add missing as NaN."""
    out = df.copy()
    for col in expected:
        if col not in out.columns:
            out[col] = pd.NA
    return out[expected]
 
 
def _predict_severity(df: pd.DataFrame) -> pd.DataFrame:
    """
    Run the severity model on `df`.
 
    Returns a DataFrame with columns:
        ACN, y_pred (0-3), p_none, p_minor, p_serious, p_fatal
        and (if 'Severity_Label' is present) y_true.
 
    Falls back to a heuristic when the real model bundle isn't loaded — this
    is the same fallback the cached path uses, so demo behaviour is consistent.
    """
    bundle = load_bundle("severity")
    model = bundle.get("model")
    preprocessor = bundle.get("preprocessor")
    metadata = bundle.get("metadata") or {}
    feature_cols: list[str] = metadata.get(
        "feature_cols", SEVERITY_SCHEMA.all_columns
    )
 
    X = _align_columns(df, feature_cols)
 
    if model is None:
        # Demo fallback — uniform-ish probabilities so the page still renders.
        n = len(df)
        rng = np.random.default_rng(seed=0)
        probs = rng.dirichlet(alpha=[2, 2, 1.5, 1], size=n)
        y_pred = probs.argmax(axis=1)
    else:
        try:
            X_proc = preprocessor.transform(X) if preprocessor is not None else X
            probs = model.predict_proba(X_proc)
            y_pred = probs.argmax(axis=1)
        except Exception as e:
            st.error(
                f"Severity model failed on uploaded data: {e}. "
                "Check that column types and value ranges match the training set."
            )
            return pd.DataFrame()
 
    out = pd.DataFrame({
        "ACN":       df.get("ACN", pd.Series(range(len(df)))).reset_index(drop=True),
        "y_pred":    y_pred,
        "p_none":    probs[:, 0],
        "p_minor":   probs[:, 1],
        "p_serious": probs[:, 2],
        "p_fatal":   probs[:, 3],
    })
    if SEVERITY_SCHEMA.target in df.columns:
        out["y_true"] = df[SEVERITY_SCHEMA.target].reset_index(drop=True)
 
    return out
 
 
def _predict_preflight(df: pd.DataFrame) -> pd.DataFrame:
    """
    Run the pre-flight risk model on `df`.
 
    Returns a DataFrame with columns:
        FL_DATE, OP_UNIQUE_CARRIER, ORIGIN, DEST, risk_score (0-1), risk_band
        and (if 'incident' is present) y_true.
    """
    bundle = load_bundle("preflight")
    model = bundle.get("model")
    preprocessor = bundle.get("preprocessor")
    metadata = bundle.get("metadata") or {}
    feature_cols: list[str] = metadata.get(
        "feature_cols", PREFLIGHT_SCHEMA.all_columns
    )
 
    X = _align_columns(df, feature_cols)
 
    if model is None:
        rng = np.random.default_rng(seed=0)
        risk = rng.beta(a=1.5, b=10, size=len(df))
    else:
        try:
            X_proc = preprocessor.transform(X) if preprocessor is not None else X
            risk = model.predict_proba(X_proc)[:, 1]
        except Exception as e:
            st.error(
                f"Pre-flight model failed on uploaded data: {e}. "
                "Check that column types and value ranges match the training set."
            )
            return pd.DataFrame()
 
    bands = pd.cut(
        risk,
        bins=[-0.001, 0.10, 0.30, 1.0],
        labels=["LOW", "MODERATE", "HIGH"],
    )
 
    out = pd.DataFrame({
        "FL_DATE":           df.get("FL_DATE", pd.NaT).reset_index(drop=True),
        "OP_UNIQUE_CARRIER": df.get("OP_UNIQUE_CARRIER", pd.NA).reset_index(drop=True),
        "ORIGIN":            df.get("ORIGIN", pd.NA).reset_index(drop=True),
        "DEST":              df.get("DEST", pd.NA).reset_index(drop=True),
        "risk_score":        risk,
        "risk_band":         bands,
    })
    if PREFLIGHT_SCHEMA.target in df.columns:
        out["y_true"] = df[PREFLIGHT_SCHEMA.target].reset_index(drop=True)
 
    return out

@st.cache_data(show_spinner=False)
def load_severity_test_predictions() -> pd.DataFrame:
    """Default page render: held-out test split + cached predictions."""
    test_path = DATA_DIR / "severity_test.parquet"
    if not test_path.exists():
        return pd.DataFrame()
    df = pd.read_parquet(test_path)
    return _predict_severity(df)
 
 
@st.cache_data(show_spinner=False)
def load_preflight_test_data() -> pd.DataFrame:
    """Default page render: held-out test split + cached predictions."""
    test_path = DATA_DIR / "preflight_test.parquet"
    if not test_path.exists():
        return pd.DataFrame()
    df = pd.read_parquet(test_path)
    return _predict_preflight(df)
 
 
# ---------------------------------------------------------------------------
# Public API — uploaded-data path (called by Manual Test feature)
# ---------------------------------------------------------------------------
def run_severity_predictions(df: pd.DataFrame) -> pd.DataFrame:
    """Run the severity pipeline on a user-uploaded DataFrame.
 
    Not cached: uploads can change between runs, and the validator already
    capped row count.
    """
    return _predict_severity(df)
 
 
def run_preflight_predictions(df: pd.DataFrame) -> pd.DataFrame:
    """Run the pre-flight pipeline on a user-uploaded DataFrame."""
    return _predict_preflight(df)
